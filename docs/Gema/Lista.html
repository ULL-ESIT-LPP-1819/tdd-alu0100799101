def pop_back
    if @n_nodos == 0
        nil
    elsif @n_nodos == 1
        valor_aux = @cola.valor
        @cola = @cola.previo
        @cabeza = @cola
        @n_nodos -= 1
        valor_aux
    else
        valor_aux = @cola.valor
        @cola = @cola.previo
        @cola.siguiente = nil
        @n_nodos -= 1
        valor_aux
    end
end

def push_back(valor)
    nodo_aux = Nodo.new(valor, nil, nil)
    if @n_nodos == 0
        @cabeza = @cola = nodo_aux
    else
        nodo_aux.previo = @cola
        @cola.siguiente = nodo_aux
        @cola = nodo_aux
    end
    @n_nodos+=1
end
